const express = require('express'); const path = require('path'); const bodyParser = require('body-parser'); const imageController = require('./imageController'); const userController = require('./userController'); const app = express(); const PORT = process.env.PORT || 3000; app.use(bodyParser.json()); app.use(express.static(path.join(__dirname, 'public'))); // Serve static files app.use('/images', imageController); app.use('/users', userController); app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); }); app.listen(PORT, () => { console.log(`Server is running on port ${PORT}`); });
